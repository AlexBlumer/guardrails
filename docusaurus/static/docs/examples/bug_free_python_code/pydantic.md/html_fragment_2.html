def longestPalindrome(s):
    n = len(s)
    dp = [[False for _ in range(n)] for _ in range(n)]
    ans = ""
    for i in range(n-1, -1, -1):
        for j in range(i, n):
            dp = s == s and (j - i < 3 or dp)
            if dp and j - i + 1 > len(ans):
                ans = s
    return ans
